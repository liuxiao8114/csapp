## p6.18

```c
struct algae_position {
  int x;
  int y;
};

struct algae_position grid[32][32];
int total_x = 0, total_y = 0;
int i, j;

for(i = 31; i >= 0; i--)
  for(j = 31; j >= 0; j--)
    total_x += grid[i][j].x;

for(i = 31; i >= 0; i--)
  for(j = 31; j >= 0; j--)
```
